## Svelte Files

### Import Order
```js
// stylesheets - first since these are exceedingly rare
import "#styles/";

// JavaScript modules (external)
import { feature } from "external";

// JavaScript modules (internal)
import { store } from "#scripts/";
import { feature } from "#scripts/";
import { utility } from "#scripts/utils/";
import type { utility } from "#scripts/types/";

// Svelte components (special)
import OneOff from "./";


// Svelte components (core)
// - sort by appearance order
import Nav from "#parts/core/";
import Header from "#parts/core/";
import Footer from "#parts/core/";
// Svelte components (UI)
// - sort alphabetical by component name
import Element from "#parts/ui/";
// Svelte components (page)
// - sort alphabetical by component name, except standard first 3
import Meta from "#parts/page";
import Breadcrumbs from "#parts/page";
import Layout from "#parts/page/";

// Javascript modules (SvelteKit)
// - sort alphabetical by path name
import { base } from "$app/paths";
```
